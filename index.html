function renderCards() { const container = document.getElementById('cardContainer'); container.innerHTML = ''; allNotes.forEach((note, index) => { const card = document.createElement('div'); // 增加更強的懸浮發光效果 card.className = "glass p-8 rounded-3xl hover:border-white/40 transition-all duration-300 group relative cursor-pointer hover:shadow-[0_0_30px_rgba(99,102,241,0.2)]"; // 根據詞性給予不同顏色 const typeColor = note.type?.includes('v.') ? 'text-red-400' : note.type?.includes('n.') ? 'text-blue-400' : 'text-slate-400'; card.innerHTML = `
${note.category} ${note.type ? `${note.type}` : ''}

${note.en}

${note.cn}

${note.example ? `

Contextual Example

"${note.example}"

` : ''} ${isQuizMode ? `
點擊卡片揭曉答案
` : ''} `; // 點擊事件：揭曉答案 card.onclick = () => { if (isQuizMode) { const cnText = document.getElementById(`cn-${index}`); const exText = document.getElementById(`ex-${index}`); if (cnText) cnText.classList.add('quiz-revealed'); if (exText) exText.classList.add('quiz-revealed'); } }; container.appendChild(card); }); }
